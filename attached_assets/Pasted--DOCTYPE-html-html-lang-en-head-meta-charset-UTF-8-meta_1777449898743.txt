<script>

// LOAD SITE (simulate dynamic site)
const siteHTML = `
<!DOCTYPE html>
<html>
<head>
<style>
body { font-family: sans-serif; text-align:center; }
button { padding:10px; margin:10px; }
</style>
</head>
<body>

<h1>🚀 My Generated Website</h1>
<p>This is your live preview.</p>

<button onclick="alert('Hello User!')">Click Me</button>

</body>
</html>
`;

document.getElementById("siteFrame").srcdoc = siteHTML;

// NAVIGATION
function goBack() {
  window.history.back();
}

// EDIT SYSTEM
function sendEdit() {
  const input = document.getElementById("editInput").value;

  if (!input) return;

  addLog("Applying change: " + input);

  // simulate change
  setTimeout(() => {
    addLog("Rebuilding...");
  }, 500);

  setTimeout(() => {
    addLog("Update complete ✅");

    // simulate new preview
    document.getElementById("siteFrame").srcdoc = `
    <html>
      <body style="font-family:sans-serif;text-align:center;">
        <h1>✨ Updated Website</h1>
        <p>${input}</p>
      </body>
    </html>
    `;
  }, 1500);

  document.getElementById("editInput").value = "";
}

// LOGS
function addLog(text) {
  const log = document.createElement("div");
  log.innerText = "• " + text;
  document.getElementById("logs").appendChild(log);
}

</script>
