<script>
const chat = document.getElementById("chat");
const logs = document.getElementById("logs");
const progressBar = document.getElementById("progressBar");
const previewFrame = document.getElementById("previewFrame");
const input = document.getElementById("prompt");

// SWITCH TABS
function switchTab(tab) {
  document.getElementById("chatTab").classList.add("hidden");
  document.getElementById("logsTab").classList.add("hidden");
  document.getElementById("previewTab").classList.add("hidden");

  document.querySelectorAll("button").forEach(b => b.classList.remove("tab-active"));

  if (tab === "chat") {
    document.getElementById("chatTab").classList.remove("hidden");
    document.querySelectorAll("button")[0].classList.add("tab-active");
  }
  if (tab === "logs") {
    document.getElementById("logsTab").classList.remove("hidden");
    document.querySelectorAll("button")[1].classList.add("tab-active");
  }
  if (tab === "preview") {
    document.getElementById("previewTab").classList.remove("hidden");
    document.querySelectorAll("button")[2].classList.add("tab-active");
  }
}

// CHAT
function addChat(text, type="user") {
  const div = document.createElement("div");
  div.className = type === "user" ? "text-right" : "text-cyan-400";
  div.innerText = text;
  chat.appendChild(div);
  chat.scrollTop = chat.scrollHeight;
}

// LOGS
function addLog(text) {
  const div = document.createElement("div");
  div.innerText = "• " + text;
  logs.appendChild(div);
  logs.scrollTop = logs.scrollHeight;
}

// PROGRESS
function setProgress(val) {
  progressBar.style.width = val + "%";
}

// SIMULATED AGENT
async function runAgent(prompt) {

  addChat(prompt, "user");

  // ANALYSIS
  addChat("Analyzing your request...", "ai");
  addLog("Detecting project type...");
  setProgress(10);
  await delay(800);

  addLog("Multi-page website detected");
  addLog("Features: auth, responsive UI");
  setProgress(25);

  // PREVIEW
  addChat("Preview ready. Switching to preview tab...", "ai");
  switchTab("preview");

  await delay(1000);

  // BUILD
  switchTab("logs");
  addChat("Building project...", "ai");

  const steps = [
    "Planning architecture",
    "Generating pages",
    "Adding styles",
    "Adding JS logic",
    "Validating structure",
    "Deploying"
  ];

  for (let i = 0; i < steps.length; i++) {
    addLog(steps[i]);
    setProgress(30 + i * 10);
    await delay(700);
  }

  // RESULT
  previewFrame.srcdoc = `
    <html>
    <body style="font-family:sans-serif;text-align:center;">
      <h1>🚀 Your App Ready</h1>
      <p>This is your generated project preview.</p>
    </body>
    </html>
  `;

  switchTab("preview");
  setProgress(100);
}

// INPUT
input.addEventListener("keydown", e => {
  if (e.key === "Enter") {
    runAgent(input.value);
    input.value = "";
  }
});

function delay(ms) {
  return new Promise(res => setTimeout(res, ms));
}
